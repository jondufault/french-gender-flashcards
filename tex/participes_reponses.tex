\section*{Exercice B --- Corrigé}

\begin{tabular}{rl}
\ex{Elle a \textcolor{accent}{\textbf{lu}} tous les livres que je lui ai \textcolor{accent}{\textbf{offerts}}. \quad{\small(accord: livres = masc. pl.)}}
\ex{La lettre que j'ai \textcolor{accent}{\textbf{écrite}} n'a jamais été \textcolor{accent}{\textbf{reçue}}. \quad{\small(accord: lettre = fém. sg.)}}
\ex{Les enfants ont \textcolor{accent}{\textbf{bu}} le jus que leur mère avait \textcolor{accent}{\textbf{fait}}. \quad{\small(accord: jus = masc. sg.)}}
\ex{La porte était \textcolor{accent}{\textbf{ouverte}} et la lumière \textcolor{accent}{\textbf{éteinte}}. \quad{\small(accord: porte/lumière = fém.)}}
\ex{Elle est \textcolor{accent}{\textbf{née}} en France et elle a \textcolor{accent}{\textbf{vécu}} à Paris. \quad{\small(née: être = accord, vécu: avoir sans COD = pas d'accord)}}
\ex{La décision qu'ils ont \textcolor{accent}{\textbf{prise}} m'a \textcolor{accent}{\textbf{surpris(e)}}. \quad{\small(accord: décision = fém., me = selon le genre du locuteur)}}
\ex{J'ai \textcolor{accent}{\textbf{dû}} partir parce que je n'ai pas \textcolor{accent}{\textbf{pu}} rester.}
\ex{Les tableaux qu'il a \textcolor{accent}{\textbf{peints}} ont \textcolor{accent}{\textbf{plu}} à tout le monde. \quad{\small(peints: accord, plu: plaire = jamais d'accord)}}
\ex{Est-ce que tu as \textcolor{accent}{\textbf{compris}} ce qu'il a \textcolor{accent}{\textbf{dit}} ?}
\ex{Nous avons \textcolor{accent}{\textbf{suivi}} le cours qu'on nous avait \textcolor{accent}{\textbf{promis}}.}
\end{tabular}
